<script>

function setActive(btn){

  document.querySelectorAll('nav button').forEach(b=>b.classList.remove('active'));

  btn.classList.add('active');

}

function toggleTheme(){

  const body=document.body;

  const btn=document.getElementById('themeBtn');

  if(body.dataset.theme==='light'){

    body.dataset.theme='dark';

    btn.textContent='Terang';

  } else {

    body.dataset.theme='light';

    btn.textContent='Gelap';

  }

}

function loadPage(page){

  setActive(document.getElementById('btn_'+page));

  if(page==='glb') return loadGLB();

  if(page==='glbb') return loadGLBB();

  if(page==='parabola') return loadParabola();

}

function loadGLB(){

  content.innerHTML=`

  <div class='card'>

    <h2>Gerak Lurus Beraturan (GLB)</h2>

    <h3>Apersepsi</h3><p>Gerak dengan kecepatan konstan pada lintasan lurus.</p>

    <h3>Pengertian</h3><p>a = 0 sehingga posisi berubah linear terhadap waktu.</p>

    <h3>Rumus</h3><p>x(t) = x₀ + v·t</p>

    <h3>Contoh Soal</h3><p>Benda bergerak v=4 m/s dari x₀=1 m. Hitung posisi setelah 6 s.</p>

    <h3>Simulasi</h3>

    <label>x₀</label><input id='x0' type='number' value='0'>

    <label>v</label><input id='v' type='number' value='3'>

    <label>Batas t</label><input id='tmax' type='number' value='10'>

    <button onclick='runGLB()'>Mulai</button>

    <div class='sim'><div class='dot' id='dot'></div></div>

  </div>`;

}

function runGLB(){

  let t=0;

  const x0=parseFloat(document.getElementById('x0').value);

  const v=parseFloat(document.getElementById('v').value);

  const tmax=parseFloat(document.getElementById('tmax').value);

  const dot=document.getElementById('dot');

  function loop(){

    t+=0.02;

    const x=x0+v*t;

    dot.style.left=(x*5)+'px';

    if(t<tmax) requestAnimationFrame(loop);

  }

  loop();

}

function loadGLBB(){

  content.innerHTML=`

  <div class='card'>

    <h2>Gerak Lurus Berubah Beraturan (GLBB)</h2>

    <h3>Apersepsi</h3><p>Kecepatan berubah teratur karena percepatan tetap.</p>

    <h3>Pengertian</h3><p>v(t)=v₀+at dan x(t)=x₀+v₀t+½at².</p>

    <h3>Rumus</h3><p>x(t)=x₀+v₀t+½at²</p>

    <h3>Contoh Soal</h3><p>x₀=1, v₀=4, a=2 → x(3)=22 m.</p>

    <h3>Simulasi</h3>

    <label>x₀</label><input id='x0b' type='number' value='0'>

    <label>v₀</label><input id='v0b' type='number' value='2'>

    <label>a</label><input id='ab' type='number' value='1'>

    <label>Batas t</label><input id='tmaxb' type='number' value='10'>

    <button onclick='runGLBB()'>Mulai</button>

    <div class='sim'><div class='dot' id='dotb'></div></div>

  </div>`;

}

function runGLBB(){

  let t=0;

  const x0=parseFloat(document.getElementById('x0b').value);

  const v0=parseFloat(document.getElementById('v0b').value);

  const a=parseFloat(document.getElementById('ab').value);

  const tmax=parseFloat(document.getElementById('tmaxb').value);

  const dot=document.getElementById('dotb');

  function loop(){

    t+=0.02;

    const x=x0+v0*t+0.5*a*t*t;

    dot.style.left=(x*5)+'px';

    if(t<tmax) requestAnimationFrame(loop);

  }

  loop();

}

function loadParabola(){

  content.innerHTML=`

  <div class='card'>

    <h2>Gerak Parabola</h2>

    <h3>Apersepsi</h3><p>Gabungan GLB pada sumbu x dan GLBB pada sumbu y.</p>

    <h3>Pengertian</h3><p>Lintasan parabola dibentuk dari gaya gravitasi.</p>

    <h3>Rumus</h3><p>x=v₀ cosθ · t<br>y=v₀ sinθ · t −½gt²</p>

    <h3>Contoh Soal</h3><p>v₀=20 m/s, θ=45°. Hitung jangkauan.</p>

    <h3>Simulasi</h3>

    <label>v₀</label><input id='v0p' type='number' value='20'>

    <label>θ</label><input id='angp' type='number' value='45'>

    <label>g</label><input id='gp' type='number' value='10'>

    <label>Batas t</label><input id='tmaxp' type='number' value='5'>

    <button onclick='runPar()'>Mulai</button>

    <div class='sim' id='simp'><div class='dot' id='dotp'></div></div>

  </div>`;

}

function runPar(){

  let t=0;

  const v0=parseFloat(document.getElementById('v0p').value);

  const ang=parseFloat(document.getElementById('angp').value)*Math.PI/180;

  const g=parseFloat(document.getElementById('gp').value);

  const tmax=parseFloat(document.getElementById('tmaxp').value);

  const dot=document.getElementById('dotp');

  const sim=document.getElementById('simp');

  function loop(){

    t+=0.02;

    const x=v0*Math.cos(ang)*t;

    const y=v0*Math.sin(ang)*t-0.5*g*t*t;

    dot.style.left=(x*5)+'px';

    dot.style.top=(100-y*5)+'px';

    if(t<tmax) requestAnimationFrame(loop);

  }

  loop();

}
